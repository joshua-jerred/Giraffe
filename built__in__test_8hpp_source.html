57 const uint16_t test_id_unsigned =

60 std::string id_unpaded = bst::string::intToHex(

62 std::string output = "0x";

171 std::lock_guard<std::mutex> lock(test_results_mutex_);
172
173 Json results = Json::object();
174 for (auto &test_case : test_cases_) {
175 results[test_case.getTestIdString()] =
176 testStatusToString(test_case.getStatus());
177 }

183private:
184 void groupRunner() {
185 setGroupStatus(TestStatus::RUNNING);
186
187 for (auto &test_case : test_cases_) {
188 test_case.startTest();
189 }
190
191 while (running_) {
192 bst::sleep(50);
193
194 const size_t MAX_TEST_TIME_MS = 1000;
195 bst::Timer timer{MAX_TEST_TIME_MS};
196 timer.reset();
197
198 // Find the first incomplete test and run it.
199 bool incomplete_test_found = false;
200 for (auto &test_case : test_cases_) {
201
202 while (!timer.isDone() &&
203 (test_case.getStatus() == TestStatus::WAITING ||
204 test_case.getStatus() == TestStatus::RUNNING)) {
205 test_case.update();
206 incomplete_test_found = true;
207 }
208 }
209
210 // All tests are complete, break the loop.
211 if (!incomplete_test_found) {
212 running_ = false;
213 break;
214 }
215 }
216
217 // Run the cleanup for each test.
218 for (auto &test_case : test_cases_) {
219 test_case.stopTest();
220 }
221
222 // Check if all tests passed.
223 bool all_tests_passed = true;
224 for (auto &test_case : test_cases_) {
225 if (test_case.getStatus() != TestStatus::PASSED) {
226 all_tests_passed = false;
227 break;
228 }
229 }
230
231 setGroupStatus(all_tests_passed ? TestStatus::PASSED : TestStatus::FAILED);
232 }
233
234 void setGroupStatus(TestStatus status) {
235 group_status_ = status;
236 }
