118  void update(IDataLink &gdl) {
119  if (!config_.isAprsTelemetrySenderEnabled()) {
120  status_ = Status::DISABLED;
121  return; // We're not enabled!
122  }
123 
124  // Check to see if metadata was requested.
125  if (metadata_send_state_ != MetadataSendState::NONE) {
126  sendMetadataStateMachine(gdl);
127  return;
128  }
129 
130  if (!new_data_available_) {
131  status_ = Status::WAITING_FOR_DATA;
132  return; // We don't have new data to send
133  }
134 
135  if (data_send_interval_stopwatch_.elapsedMilliseconds() <
136  config_.getAprsTelemetryDataIntervalMs()) {
137  status_ = Status::SEND_DATA_DELAY;
138  return; // Not time to send yet
139  }
140 
141  status_ = Status::SENDING_DATA;
142 
143  // Reset the interval stopwatch but don't stop it.
144  constexpr bool STOP_ON_RESET = false;
145  data_send_interval_stopwatch_.reset(STOP_ON_RESET);
146 
147  telemetry_data_lock_.lock();
148  uint16_t sequence_number = getNextSequenceNumber();
149  telemetry_data_.setSequenceNumber(sequence_number);
150  bool sent = gdl.sendTelemetryData(telemetry_data_, sequence_number);
151  telemetry_data_lock_.unlock();
152 
153  if (sent) {
154  new_data_available_ = false;
155  num_data_packets_added_to_queue_++;
156  } else {
158  std::cout << "Failed to send telemetry data" << std::endl;
159  }
160  }

208  metadata_send_state_ = MetadataSendState::UNITS_LABELS;
209  break;

212  metadata_send_state_ = MetadataSendState::COEFFICIENTS;
213  break;
214  case MetadataSendState::COEFFICIENTS:
215  gdl.sendTelemetryCoefficients(telemetry_data_);
216  metadata_send_state_ = MetadataSendState::BIT_SENSE;
217  break;
218  case MetadataSendState::BIT_SENSE:
219  gdl.sendTelemetryBitSense(telemetry_data_);
220  metadata_send_state_ = MetadataSendState::NONE;
