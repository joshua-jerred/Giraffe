102 void log(const LoggerLevel level, const std::string &msg) {
103
104 // don't log if the level of this item is lower than the level to log
105 if (static_cast<std::underlying_type_t<LoggerLevel>>(level) <
106 static_cast<std::underlying_type_t<LoggerLevel>>(level_to_log_)) {
107 return;
108 }
109
110 LoggerEntry entry{level, msg, std::chrono::system_clock::now()};
111 if (log_queue_.size() > LOG_SIZE) {
112 log_queue_.pop();
113 }
114 log_queue_.push(entry);
115
116 if (print_to_stdout_ || log_to_file_) {
117 std::string log_msg =
118 fmt::format("{} [{}] : {}\n", fmt::format("{:%H:%M:%S}", entry.time),
119 entry.getLevelString(), entry.msg);
120
121 if (print_to_stdout_) {
122 std::cout << log_msg << std::flush;
123 }
124 if (log_to_file_) {
125 // write to file
126 bst::filesystem::appendToFile(log_file_path_, log_msg, false);
127 }
128 }
129 }
