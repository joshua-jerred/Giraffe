47template <class T> class Stream {
48public:
49 Stream() : packet_queue_() {
50 }
51
52 void addPacket(T packet) {
53 stream_mutex_.lock();
54
55 packet_queue_.push(packet);
56 current_packets_++;
57 total_packets_++;
58 stream_mutex_.unlock();
59 }
60
61 bool getPacket(T &packet) {
62 std::lock_guard<std::mutex> lock(stream_mutex_);
63
64 if (packet_queue_.empty()) {
65 return false;
66 }
67
68 packet = packet_queue_.front();
69 packet_queue_.pop();
70 current_packets_--;
71
72 return true;
73 }
74
75 int getNumPackets() { // not atomic as this is used for stream processing
76 std::lock_guard<std::mutex> lock(stream_mutex_);
77 return current_packets_;
78 }
79
80 int getTotalPackets() { // atomic
81 return total_packets_;
82 }
83
84 void reset() {
85 std::lock_guard<std::mutex> lock(stream_mutex_);
86 current_packets_ = 0;
87 total_packets_ = 0;
88 packet_queue_ = std::queue<T>();
89 }
90
91private:
92 std::mutex stream_mutex_ = std::mutex();
93 std::queue<T> packet_queue_;
94
95 int current_packets_ = 0;
96 std::atomic<int> total_packets_ = 0;
97};
