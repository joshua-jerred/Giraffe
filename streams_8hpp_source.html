52  void addPacket(T packet) {

54

57  total_packets_++;
58  stream_mutex_.unlock();

61  bool getPacket(T &packet) {

63 
64  if (packet_queue_.empty()) {
65  return false;
66  }
67 
68  packet = packet_queue_.front();
69  packet_queue_.pop();
70  current_packets_--;
71 
72  return true;
73  }
