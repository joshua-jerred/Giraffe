86 void processStream(data::Stream<T> &stream,
87 data::blocks::StreamsStats::StreamStats &stats,
88 std::function<void(const T &)> parse_function) {
89 // Get the number of packets currently in the stream and the total number
90 // that have gone through the stream.
91 int packet_count = stream.getNumPackets();
92 int total_packets = stream.getTotalPackets();
93
94 T packet;
95 bool first = true;
96 int delay_ms = 0; // used to calculate the processing delay
97 for (int i = 0; i < packet_count; i++) {
98 bool got_packet = stream.getPacket(packet);
99 if (!got_packet) { // this should never happen
100 return;
101 }
102 // Calculate the delay between now and when the oldest packet was created
103 if (first) {
104 delay_ms = bst::clck::millisecondsElapsed(packet.created_time);
105 first = false;
106 }
107
108 // Pass to the packet parsing function
109 parse_function(packet);
110 }
111
112 // Update the stream stats for this stream
113 stats.current_packets = packet_count;
114 stats.total_packets = total_packets;
115 stats.processing_delay_ms = delay_ms;
116 }
