23template <class T> class Block {
24public:
25 void set(T data) {
26 std::lock_guard<std::mutex> lock(mutex_);
27 data_ = data;
28 last_updated_ = bst::clck::now();
29 }
30
31 T get() {
32 std::lock_guard<std::mutex> lock(mutex_);
33 return data_;
34 }
35
36 int getAgeMs() {
37 std::lock_guard<std::mutex> lock(mutex_);
38 return bst::clck::millisecondsElapsed(last_updated_);
39 }
40
41 int getAgeSec() {
42 std::lock_guard<std::mutex> lock(mutex_);
43 return bst::clck::secondsElapsed(last_updated_);
44 }
45
46private:
