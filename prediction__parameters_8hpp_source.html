220 void updateParameter(Parameter::Id id, double newValue, bool valid = true,
221 bool digital = false, bool digitalValue = false) {
222 giraffe_assert(id != Parameter::Id::NUM_PARAMETERS);
223
224 // If already marked invalid or the validation fails, apply the invalid
225 // filter and mark the parameter as invalid.
226 if (!valid || !validateParameter(id, newValue)) {
227 valid = false;
228 newValue = applyInvalidFilter(id, newValue);
229 }
230
231 parameters_.at(static_cast<uint8_t>(id)).is_valid = valid;
232 if (digital) {
233 parameters_.at(static_cast<uint8_t>(id)).binary_value = digitalValue;
234 } else {
235 parameters_.at(static_cast<uint8_t>(id)).value = newValue;
236 }
237 }

242 bool validateParameter(Parameter::Id id, double value) {
243 auto param = parameters_.at(static_cast<uint8_t>(id));
244
245 switch (param.validation) {
246 case Parameter::Validation::NONE:
247 return true;
248 case Parameter::Validation::IN_RANGE:
249 return value >= param.min && value <= param.max;
250 default:
251 giraffe_assert(false);
252 return false;
253 }
254
255 giraffe_assert(false);
256 return false;
257 }
258
264 double applyInvalidFilter(Parameter::Id id, double value) {
265 auto param = parameters_.at(static_cast<uint8_t>(id));
266
267 switch (param.invalidate_filter) {
268 case Parameter::InvalidateFilter::ALWAYS_UPDATE:
269 return value;
270 case Parameter::InvalidateFilter::DO_NOT_UPDATE:
271 return param.value;
272 case Parameter::InvalidateFilter::ZERO_IF_INVALID:
273 return 0.0;
274 case Parameter::InvalidateFilter::CLAMP_IN_RANGE:
275 return std::clamp(value, param.min, param.max);
276 default:
277 giraffe_assert(false);
278 return 0.0;
279 }
280
281 giraffe_assert(false);
282 return 0.0;
283 }
