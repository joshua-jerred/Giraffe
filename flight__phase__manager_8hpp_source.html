106  bool flightPhaseUpdate(FlightPhase new_phase) {
107  FlightPhase current_phase = flight_phase_;
108 
109  bool change_allowed = true;
110  bool unexpected_change = false;
111 
112  // Alert when the phase is unknown
113  if (new_phase == FlightPhase::UNKNOWN) {
114  // Suppress the error if we are in pre-launch.
115  if (current_phase == FlightPhase::PRE_LAUNCH) {
116  return false;
117  }
118 
119  // Suppress the error if we have already reported it.
120  if (unknown_phase_error_reported_) {
121  return false;
122  }
123 
124  logger_.error(DiagnosticId::FLIGHT_RUNNER_flightPhaseUnknown);
125  unknown_phase_error_reported_ = true;
126  return false;
127  } else {
128  unknown_phase_error_reported_ = false;
129  }
130 
131  // Staying in the same phase.
132  if (new_phase == current_phase) {
133  phase_valid_watchdog_.reset();
134  return false;
135  }
136 
137  static_assert(static_cast<uint8_t>(FlightPhase::NUM_PHASES) == 6,
138  "Update the switch statement");
139  switch (current_phase) {
140  case FlightPhase::UNKNOWN: {
141  // The transition from unknown to launch could be dangerous. It would be
142  // safer to go to recovery instead. Edge Cases: Out of Disk Space,
143  // FlightSystem restart mid-flight or on recovery.
144  if (new_phase == FlightPhase::LAUNCH) {
145  unexpected_change = true;
146  return flightPhaseUpdate(FlightPhase::RECOVERY);
147  }
148  } break;
149  case FlightPhase::PRE_LAUNCH: {
150  if (new_phase == FlightPhase::LAUNCH) {
151  // The transition from pre-launch to launch can only be triggered by the
152  // user. If we are set to pre-launch, the phase predictor will report
153  // launch.
154  phase_valid_watchdog_.reset();
155  return false;
156  }
157 
158  // Entering the pre-launch phase is a manual process. However, safety wise
159  // we should be able to automatically exit to ascent.
160  if (new_phase != FlightPhase::ASCENT) {
161  change_allowed = false;
162  }
163 
164  unexpected_change = true;
165  } break;
166  case FlightPhase::LAUNCH: {
167  // We expect to go to ASCENT.
168  if (new_phase != FlightPhase::ASCENT) {
169  unexpected_change = true;
170  }
171 
172  // Allow for transitions forward only.
173  if (new_phase != FlightPhase::ASCENT &&
174  new_phase != FlightPhase::DESCENT &&
175  new_phase != FlightPhase::RECOVERY) {
176  change_allowed = false;
177  }
178  } break;
179  case FlightPhase::ASCENT:
180  // We expect to go to DESCENT.
181  if (new_phase != FlightPhase::DESCENT) {
182  unexpected_change = true;
183  }
184 
185  // If we are suddenly predicting launch, we should go to recovery.
186  if (new_phase == FlightPhase::LAUNCH) {
187  forceRecovery("in ascent");
188  return true;
189  }
190 
191  // Allow for transitions forward only.
192  if (new_phase != FlightPhase::DESCENT &&
193  new_phase != FlightPhase::RECOVERY) {
194  change_allowed = false;
195  }
196  break;
197  case FlightPhase::DESCENT: {
198  if (new_phase != FlightPhase::RECOVERY) {
199  unexpected_change = true; // We expect to go to RECOVERY.
200  }
201 
202  // If we are suddenly predicting launch, we should go to recovery.
203  if (new_phase == FlightPhase::LAUNCH) {
204  forceRecovery("in descent");
205  return true;
206  }
207 
208  // Allow for transition back to ascent or forward to recovery.
209  if (new_phase != FlightPhase::ASCENT &&
210  new_phase != FlightPhase::RECOVERY) {
211  change_allowed = false;
212  }
213  } break;
214  case FlightPhase::RECOVERY: {
215  // We expect to stay in RECOVERY.
216  unexpected_change = true;
217 
218  // Allow for transition back to ascent or descent.
219  if (new_phase != FlightPhase::ASCENT &&
220  new_phase != FlightPhase::DESCENT) {
221  change_allowed = false;
222  }
223  } break;
224  default:
225  change_allowed = false;
226  break;
227  }
228 
229  if (!change_allowed) {
230  reportPhaseChangeInvalid(current_phase, new_phase, "change not allowed");
231  return false;
232  }
233 
234  if (unexpected_change) {
235  logger_.error(DiagnosticId::FLIGHT_RUNNER_phaseChangeUnexpected,
236  util::to_string(current_phase) + " -> " +
237  util::to_string(new_phase));
238  }
239 
240  phase_valid_watchdog_.reset();
241  setFlightPhase(new_phase);
242 
243  return true;
244  }
